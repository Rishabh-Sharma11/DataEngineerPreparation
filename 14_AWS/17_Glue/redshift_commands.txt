CREATE TABLE IF NOT EXISTS public.sales_data (
    ORDERNUMBER INT,
    QUANTITYORDERED INT,
    PRICEEACH DOUBLE PRECISION,
    ORDERLINENUMBER INT,
    SALES DOUBLE PRECISION,
    ORDERDATE VARCHAR(50),
    STATUS VARCHAR(50), 
    QTR_ID INT,
    MONTH_ID INT,
    YEAR_ID INT
)
DISTSTYLE KEY
DISTKEY (YEAR_ID)
SORTKEY (YEAR_ID, MONTH_ID);
